components/ScoreChip.tsx

import clsx from "clsx";

export function ScoreChip({ score }: { score: number }) {
  const band =
    score >= 80 ? "bg-green-500/15 text-green-300" :
    score >= 60 ? "bg-yellow-500/15 text-yellow-300" :
                  "bg-red-500/15 text-red-300";
  return (
    <span
      aria-label={`Impact score ${score}`}
      className={clsx("inline-flex items-center rounded-full px-2.5 py-1 text-sm font-medium", band)}
    >
      {score}
    </span>
  );
}
